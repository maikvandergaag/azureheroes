<script>
(function() {
  var current = 0;
  var total = 33;
  var autoTimer;

  function showSlide(n) {
    var slides = document.querySelectorAll('#picksCarousel .picks-slide');
    current = (n + total) % total;
    slides.forEach(function(s, i) { s.classList.toggle('active', i === current); });
    document.getElementById('picksCounter').textContent = (current + 1) + ' / ' + total;
    document.querySelectorAll('#picksDots .picks-dot').forEach(function(d, i) {
      d.classList.toggle('active', i === current);
    });
  }

  function resetAuto() {
    clearInterval(autoTimer);
    autoTimer = setInterval(function() { showSlide(current + 1); }, 4000);
  }

  window.picksMove = function(dir) {
    resetAuto();
    showSlide(current + dir);
  };

  document.addEventListener('DOMContentLoaded', function() {
    var dotsEl = document.getElementById('picksDots');
    for (var i = 0; i < total; i++) {
      (function(idx) {
        var dot = document.createElement('span');
        dot.className = 'picks-dot';
        dot.onclick = function() { resetAuto(); showSlide(idx); };
        dotsEl.appendChild(dot);
      })(i);
    }
    showSlide(0);
    resetAuto();

    document.addEventListener('keydown', function(e) {
      if (e.key === 'ArrowLeft') picksMove(-1);
      if (e.key === 'ArrowRight') picksMove(1);
    });
  });
})();
</script>
